class Solution {
   public int fibByDp(int n,int arr[]){
    if(n==0 || n==1) return n;
    if(arr[n]!=-1) return arr[n];
    arr[n]=fibByDp(n-1,arr) + fibByDp(n-2,arr);

   }

    public int fib(int n) {
    int[] arr=new int[n+1];
    Arrays.fill(arr,-1);
    int ans =fibByDp(n,arr);
    return ans;
